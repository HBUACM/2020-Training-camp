int checknum(char* s)//判断是否是数，是则返回num，不是则返回-1
{
    int sum = 0;
    for (int i = 0; i < strlen(s); i++)
    {
        if (s[i] >= '0' && s[i] <= '9')
            sum = sum * 10 + s[i] - '0';
        else
            return -1;
    }
    if (sum >= 1 && sum <= 1000)
        return sum;
    else
        return -1;
}
