int parseint(string s) {
        int ret = 0;
        if (s.length() > 10)
                return -1;
        for (int i = 0; i < s.length(); i++) {
                if (!isdigit(s[i]))
                        return -1;
                else
                        ret = ret * 10 + s[i] - '0';
        }
        if (ret > 1000 || ret == 0)
                return -1;
        return ret;
}
