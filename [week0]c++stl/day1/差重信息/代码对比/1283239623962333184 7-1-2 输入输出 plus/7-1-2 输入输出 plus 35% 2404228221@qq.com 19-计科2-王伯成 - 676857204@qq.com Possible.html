#include <iostream>
#include <string>

using namespace std;

bool if_dig(string& str) {
        int len = str.size();
        if (len > 4) {
                return false;
        }
        for (int i = 0; i < len; i++) {
                if (!isdigit(str[i])) {
                        return false;
                }
        }
        if (stoi(str) > 1000|| stoi(str)==0) {
                return false;
        }
        return true;
}
